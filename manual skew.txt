import numpy as np
data = [12, 45, 33, 27, 58, 61, 39, 50, 48, 41]
mean = np.mean(data)
median = np.median(data)
std_dev = np.std(data, ddof=1)
skew = 3 * (mean - median) / std_dev
print("Mean:", mean)
print("Median:", median)
print("Standard Deviation:", std_dev)
print("Skewness:", skew)
if skew > 0:
    print("\n\n Data is Positively Skewed (Right Skewed)")
elif skew < 0:
    print("\n \n Data is Negatively Skewed (Left Skewed)")
else:
    print("\n \n Data is Symmetric")
